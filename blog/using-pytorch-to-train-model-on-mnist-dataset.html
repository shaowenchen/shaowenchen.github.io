import torch
import torch.nn as nn
import torch.nn.functional as F
import torch.optim as optim
from torchvision import datasets, transforms

# å®šä¹‰è¶…å‚æ•°
BATCH_SIZE = 512
EPOCHS = 20
LEARNING_RATE = 1e-3
DEVICE = torch.device("cuda" if torch.cuda.is_available() else "cpu")

# æ•°æ®åŠ è½½å’Œé¢„å¤„ç†ï¼Œåœ¨ data ç›®å½•ä¸‹è½½ MNIST æ•°æ®é›†
train_loader = torch.utils.data.DataLoader(
    datasets.MNIST('data', train=True, download=True,
                   transform=transforms.Compose([
                       transforms.RandomRotation(10),  # éšæœºæ—‹è½¬ï¼Œæé«˜æ¨¡å‹çš„æ³›åŒ–èƒ½åŠ›
                       transforms.RandomAffine(0, shear=10, scale=(0.8, 1.2)),  # ä»¿å°„å˜æ¢ï¼Œæé«˜æ¨¡å‹çš„æ³›åŒ–èƒ½åŠ›
                       transforms.ToTensor(),
                       transforms.Normalize((0.1307,), (0.3081,))
                   ])),
    batch_size=BATCH_SIZE, shuffle=True)

test_loader = torch.utils.data.DataLoader(
    datasets.MNIST('data', train=False,
                   transform=transforms.Compose([
                       transforms.ToTensor(),
                       transforms.Normalize((0.1307,), (0.3081,))
                   ])),
    batch_size=BATCH_SIZE, shuffle=False)

# å®šä¹‰æ¨¡å‹çš„ç»“æ„ï¼Œè¿™é‡Œæ˜¯ä¸€ä¸ªç®€å•çš„å·ç§¯ç¥ç»ç½‘ç»œ
class ConvNet(nn.Module):
    def __init__(self):
        super(ConvNet, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, 3)
        self.conv2 = nn.Conv2d(32, 64, 3)
        self.dropout1 = nn.Dropout(0.25)
        self.fc1 = nn.Linear(64 * 5 * 5, 128)
        self.dropout2 = nn.Dropout(0.5)
        self.fc2 = nn.Linear(128, 10)

    def forward(self, x):
        x = F.relu(self.conv1(x))
        x = F.max_pool2d(x, 2)
        x = F.relu(self.conv2(x))
        x = F.max_pool2d(x, 2)
        x = self.dropout1(x)
        x = x.view(-1, 64 * 5 * 5)
        x = F.relu(self.fc1(x))
        x = self.dropout2(x)
        x = self.fc2(x)
        return F.log_softmax(x, dim=1)

# åˆå§‹åŒ–æ¨¡å‹å’Œä¼˜åŒ–å™¨
model = ConvNet().to(DEVICE)
optimizer = optim.Adam(model.parameters(), lr=LEARNING_RATE)
scheduler = torch.optim.lr_scheduler.StepLR(optimizer, step_size=5, gamma=0.5)  # å­¦ä¹ ç‡è¡°å‡

# è®­ç»ƒå‡½æ•°
def train(model, device, train_loader, optimizer, epoch):
    model.train()
    for batch_idx, (data, target) in enumerate(train_loader):
        data, target = data.to(device), target.to(device)
        optimizer.zero_grad()
        output = model(data)
        loss = F.nll_loss(output, target)
        loss.backward()
        optimizer.step()
        if (batch_idx + 1) % 30 == 0:
            print(f"Train Epoch: {epoch} [{batch_idx * len(data)}/{len(train_loader.dataset)} "
                  f"({100. * batch_idx / len(train_loader):.0f}%)]\tLoss: {loss.item():.6f}")

# æµ‹è¯•å‡½æ•°
def test(model, device, test_loader):
    model.eval()
    test_loss = 0
    correct = 0
    with torch.no_grad():
        for data, target in test_loader:
            data, target = data.to(device), target.to(device)
            output = model(data)
            test_loss += F.nll_loss(output, target, reduction='sum').item()
            pred = output.argmax(dim=1, keepdim=True)
            correct += pred.eq(target.view_as(pred)).sum().item()
    test_loss /= len(test_loader.dataset)
    accuracy = 100. * correct / len(test_loader.dataset)
    print(f"\nTest set: Average loss: {test_loss:.4f}, Accuracy: {correct}/{len(test_loader.dataset)} ({accuracy:.2f}%)\n")

# è®­ç»ƒå’Œæµ‹è¯•å¾ªç¯
for epoch in range(1, EPOCHS + 1):
    train(model, DEVICE, train_loader, optimizer, epoch)
    test(model, DEVICE, test_loader)
    scheduler.step()  # æ›´æ–°å­¦ä¹ ç‡

# ä¿å­˜æ¨¡å‹
torch.save(model.state_dict(), "mnist_cnn.pth")
print("Model saved to mnist_cnn.pth")

class ConvNet(nn.Module):

        super(ConvNet, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, 3)
        self.conv2 = nn.Conv2d(32, 64, 3)
        self.dropout1 = nn.Dropout(0.25)
        self.fc1 = nn.Linear(64 * 5 * 5, 128)
        self.dropout2 = nn.Dropout(0.5)
        self.fc2 = nn.Linear(128, 10)

    def forward(self, x):
        x = F.relu(self.conv1(x))
        x = F.max_pool2d(x, 2)
        x = F.relu(self.conv2(x))
        x = F.max_pool2d(x, 2)
        x = self.dropout1(x)
        x = x.view(-1, 64 * 5 * 5)
        x = F.relu(self.fc1(x))
        x = self.dropout2(x)
        x = self.fc2(x)
        return F.log_softmax(x, dim=1)

# å›¾ç‰‡é¢„å¤„ç†
def preprocess_image(image_path):
    transform = transforms.Compose([
        transforms.Grayscale(num_output_channels=1),  # ç¡®ä¿æ˜¯å•é€šé“ï¼ˆç°åº¦å›¾ï¼‰
        transforms.Resize((28, 28)),  # è°ƒæ•´å¤§å°ä¸º28x28
        transforms.ToTensor(),  # è½¬ä¸ºTensor
        transforms.Normalize((0.1307,), (0.3081,))  # æ ‡å‡†åŒ–
    ])
    image = Image.open(image_path).convert('RGB')  # ç¡®ä¿å›¾ç‰‡ä¸ºRGBæ¨¡å¼ï¼Œé¿å…æŠ¥é”™
    return transform(image).unsqueeze(0)  # æ·»åŠ æ‰¹æ¬¡ç»´åº¦

def predict(image_path):

        image_tensor = preprocess_image(image_path).to(DEVICE)

            pred = output.argmax(dim=1, keepdim=True)

    except Exception as e:
        print(f"Error processing image '{image_path}': {e}")
